ubuntu 16.04 준비

php7.1 설치

sudo apt-get install software-properties-common
sudo add-apt-repository ppa:ondrej/php
sudo apt update

// 여긴 필요한 것만 사용하도록 수정
sudo apt install php7.1-fpm php7.1-common php7.1-mbstring php7.1-xmlrpc php7.1-soap php7.1-gd php7.1-xml php7.1-intl php7.1-mysql php7.1-cli php7.1-mcrypt php7.1-zip php7.1-curl

extention 설치

php amfext 설치

git clone https://github.com/emilkm/amfext.git
cd amfext
phpize7.1
sudo configure --enabled-amf && sudo make install
 경로 수정
echo "extension = amf.so" | sudo tee -a /etc/php/7.1/mods-available/amf.ini

sudo ln -s /etc/php/7.1/mods-available/amf.ini /etc/php/7.1/fpm/conf.d/30-amf.ini
sudo ln -s /etc/php/7.1/mods-available/amf.ini /etc/php/7.1/cli/conf.d/30-amf.ini

apc(u) 설치

sudo apt-get install php7.1-apcu -y

nginx 설치

sudo apt install nginx
sudo systemctl stop nginx.service
sudo systemctl start nginx.service
sudo systemctl enable nginx.service
sudo  /lib/systemd/systemd-sysv-install enable nginx
